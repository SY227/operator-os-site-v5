## Environment variables

Copy `.env.example` to `.env.local` and configure as needed.

Required only for persistent storage/providers:

- `LEADS_PROVIDER` (`console|webhook|both`)
- `LEADS_WEBHOOK_URL` (required if using webhook leads)
- `ANALYTICS_PROVIDER` (`console|webhook|both`)
- `ANALYTICS_WEBHOOK_URL` (required if using webhook analytics)
- `ANALYTICS_CONSOLE_LOGS` (`true|false`)
- `NEXT_PUBLIC_ANALYTICS_DEBUG` (`true|false`)

## How to add content quickly

### Add a new product
1. Add a new object to `content/products.json` with a unique `slug`
2. Include `primaryCta.label` + `primaryCta.href`
3. Product auto-appears on listing and gets detail route at `/products/[slug]`

### Add a new free tool
1. Add object to `content/free-tools.json`
2. Keep a unique `slug`
3. Tool auto-appears and gets detail route at `/free-tools/[slug]`

### Add a new article
1. Add object to `content/articles.json`
2. Include `body` array for detail page paragraphs
3. Article auto-appears and gets detail route at `/articles/[slug]`

# OPERATIONS.md

Operational runbook for OpenClaw maintaining Operator OS Site V2.

## 1) Recurring content update workflow

Frequency: weekly

1. Pull latest repo changes
2. Edit only content JSON files under `/content` for normal content updates
3. Validate JSON syntax
4. Run:
   - `npm run lint`
   - `npm run build`
5. Deploy to Vercel

### OpenClaw safe-edit scope
- ✅ Safe by default: `/content/*.json`, `README.md`, `OPERATIONS.md`
- ⚠️ Requires caution: `src/app/**`, `src/lib/**`, API handlers

---

## 2) Recurring lead triage workflow

Frequency: daily or every 2 days

1. Check lead destination (webhook/CRM) for new records from types:
   - newsletter
   - contact
   - diagnostic
   - free-tool-request
2. Prioritize contact + diagnostic leads first
3. Follow up with:
   - score/tier context from diagnostic submissions
   - requested tool/product context from CTA and slug fields
4. Track response SLA (target: <24h business day)

---

## 3) Recurring analytics review workflow

Frequency: weekly

Review key events:
- `page_view`
- `cta_click`
- `newsletter_submit`
- `contact_submit`
- `diagnostic_start`
- `diagnostic_complete`
- `free_tool_request`
- `product_cta_click`

Check:
1. Top pages by page view
2. CTA click-through by section
3. Diagnostic start-to-complete ratio
4. Free tool request volume by `toolSlug`
5. Product CTA engagement by `productSlug`

---

## 4) What OpenClaw should NOT edit without explicit request

- Core lead/analytics provider contracts in `src/lib/leads/types.ts` and `src/lib/analytics/types.ts`
- API response schemas in `src/app/api/**/route.ts`
- Legal wording in `/privacy` and `/terms` beyond minor copy formatting unless requested
- Tailwind/core layout architecture unless required for bug fixes

import { NextResponse } from "next/server";

import { trackServerEvent } from "@/lib/analytics/server";

export async function POST(request: Request) {

    const body = await request.json();
    const email = normalizeString(body?.email).toLowerCase();
    const sourcePath = normalizeString(body?.sourcePath) || "/";

    if (!isValidEmail(email)) {
      return NextResponse.json(

        sourcePath,
      },
    });

    await trackServerEvent({
      name: "newsletter_submit",
      path: sourcePath,
      properties: { emailDomain: email.split("@")[1] ?? "unknown" },
    });

    return NextResponse.json({
      ok: true,
      message: "You’re on the list. Expect practical systems and workflows soon.",
      warning: leadResult.warning,
      storage: {
        persisted: leadResult.persisted,
        providers: leadResult.results,

    });

      {
        ok: false,
        error: {
          code: "BAD_REQUEST",
          message: "Unable to process newsletter signup payload.",
        },
      },

import { NextResponse } from "next/server";
import { scoreDiagnostic } from "@/lib/diagnostic";
import { submitLead } from "@/lib/leads";
import { trackServerEvent } from "@/lib/analytics/server";
import { isObject, isValidEmail, normalizeString } from "@/lib/validation";

export async function POST(request: Request) {
  try {
    const body = await request.json();

    const name = normalizeString(body?.name) || undefined;
    const email = normalizeString(body?.email).toLowerCase();
    const sourcePath = normalizeString(body?.sourcePath) || "/get-started";

    if (email && !isValidEmail(email)) {
      return NextResponse.json(
        {
          ok: false,
          error: {
            code: "INVALID_EMAIL",
            message: "If provided, email must be valid.",
            fields: ["email"],
          },
        },
        { status: 400 },
      );
    }

    if (!isObject(body?.answers)) {
      return NextResponse.json(
        {
          ok: false,
          error: {
            code: "INVALID_ANSWERS",
            message: "Answers must be an object keyed by diagnostic question id.",
            fields: ["answers"],
          },
        },
        { status: 400 },
      );
    }

    const answers = Object.entries(body.answers).reduce<Record<string, number>>(
      (acc, [key, value]) => {
        if (typeof value === "number" && Number.isFinite(value)) {
          acc[key] = value;
        }
        return acc;
      },
      {},
    );

    const result = await scoreDiagnostic(answers);

    const leadResult = await submitLead({
      type: "diagnostic",
      name,
      email: email || undefined,
      sourcePath,
      createdAt: new Date().toISOString(),
      payload: {
        name,
        email: email || undefined,
        answers,
        result,
      },
    });

    await trackServerEvent({
      name: "diagnostic_complete",
      path: sourcePath,
      properties: {
        score: result.totalScore,
        maxScore: result.maxScore,
        tier: result.maturityTierId,
        recommendedProductSlug: result.recommendedProductSlug,
      },
    });

    return NextResponse.json({
      ok: true,
      result,
      warning: leadResult.warning,
      storage: {
        persisted: leadResult.persisted,
        providers: leadResult.results,
      },
    });
  } catch {
    return NextResponse.json(
      {
        ok: false,
        error: {
          code: "BAD_REQUEST",
          message: "Unable to process diagnostic payload.",
        },
      },
      { status: 400 },
    );
  }
}

    if (!isNonEmptyString(name) || !allowedEventNames.has(name)) {
      return NextResponse.json(
        {
          ok: false,
          error: {
            code: "INVALID_EVENT_NAME",
            message: "Unsupported analytics event name.",
          },
        },
        { status: 400 },
      );
    }

    await trackServerEvent({
      name: name as
        | "page_view"
        | "cta_click"
        | "newsletter_submit"
        | "contact_submit"
        | "diagnostic_start"
        | "diagnostic_complete"
        | "free_tool_request"
        | "product_cta_click",
      path,
      properties:
        body?.properties && typeof body.properties === "object"
          ? body.properties
          : undefined,
      timestamp: normalizeString(body?.timestamp) || undefined,
    });

    return NextResponse.json({ ok: true });
  } catch {
    return NextResponse.json(
      {
        ok: false,
        error: {
          code: "BAD_REQUEST",
          message: "Invalid analytics payload.",
        },
      },
      { status: 400 },
    );
  }
}

import Link, { LinkProps } from "next/link";
import { ReactNode } from "react";
import { trackClientEvent } from "@/lib/analytics/client";
import { AnalyticsEventName } from "@/lib/analytics/types";

type TrackedLinkProps = LinkProps & {
  className?: string;
  children: ReactNode;
  eventName?: AnalyticsEventName;
  eventProps?: Record<string, unknown>;
  eventPayload?: Record<string, unknown>; // backwards-compatible alias
};

export default function TrackedLink({
  className,
  children,
  eventName = "cta_click",
  eventProps,
  eventPayload,
  ...linkProps
}: TrackedLinkProps) {
  const properties = eventProps ?? eventPayload;

  return (
    <Link
      {...linkProps}
      className={className}
      onClick={() => {
        const targetPath =
          typeof linkProps.href === "string"
            ? linkProps.href
            : linkProps.href.pathname ?? undefined;

        trackClientEvent({
          name: eventName,
          path: targetPath,
          properties,
        });
      }}
    >
      {children}
    </Link>
  );
}

function hasProjectMarkers(dir: string) {
  const packageJsonPath = path.join(dir, "package.json");
  const contentPath = path.join(dir, "content");

  if (!existsSync(packageJsonPath) || !existsSync(contentPath)) {
    return false;
  }

  try {
    const pkg = JSON.parse(readFileSync(packageJsonPath, "utf8")) as {
      name?: string;
    };
    return typeof pkg.name === "string" && pkg.name.includes("operator-os-site");
  } catch {
    return false;
  }
}

function resolveProjectRoot() {
  const candidates = new Set<string>();

  candidates.add(process.cwd());
  if (process.env.PWD) {
    candidates.add(process.env.PWD);
  }

  for (const arg of process.argv) {
    if (!arg || arg.startsWith("-")) {
      continue;
    }

    const absolute = path.resolve(arg);
    candidates.add(absolute);
    candidates.add(path.dirname(absolute));
  }

  candidates.add(path.join(process.cwd(), "operator-os-site-v2"));

  for (const candidate of candidates) {
    if (hasProjectMarkers(candidate)) {
      return candidate;
    }
  }

  return process.cwd();
}

const PROJECT_ROOT = resolveProjectRoot();
const CONTENT_DIR = path.join(PROJECT_ROOT, "content");
const cache = new Map<string, unknown>();

async function loadContentFile<T>(fileName: string): Promise<T> {
  if (cache.has(fileName)) {
    return cache.get(fileName) as T;
  }

  const fullPath = path.join(CONTENT_DIR, fileName);
  const raw = await readFile(fullPath, "utf8");
  const parsed = JSON.parse(raw) as T;
  cache.set(fileName, parsed);
  return parsed;
}

export function clearContentCache() {
  cache.clear();
}

import { LeadProvider, LeadRecord, LeadStoreResult } from "./types";

function getProviders(): LeadProvider[] {
  const mode = process.env.LEADS_PROVIDER ?? "console";

  if (mode === "webhook") {
    return webhookLeadProvider ? [webhookLeadProvider] : [consoleLeadProvider];
  }

  if (mode === "both") {
    return webhookLeadProvider
      ? [consoleLeadProvider, webhookLeadProvider]
      : [consoleLeadProvider];
  }

  return [consoleLeadProvider];
}

export async function submitLead(record: LeadRecord) {
  const providers = getProviders();
  const results: LeadStoreResult[] = [];

  for (const provider of providers) {
    results.push(await provider.storeLead(record));
  }

  const persisted = results.some((result) => result.persisted);
  const warning = persisted
    ? undefined
    : "No persistent lead destination configured. Set LEADS_WEBHOOK_URL (and LEADS_PROVIDER=webhook or both).";

  return {
    ok: results.some((result) => result.ok),
    persisted,
    warning,
    results,
  };
}

            ok: true,

          };

            provider: "webhook",
            ok: false,
            persisted: false,
            message:
              error instanceof Error ? error.message : "Unknown webhook error",
          };
        }
      },
    }
  : null;

import { AnalyticsEvent } from "./types";

export async function trackClientEvent(event: AnalyticsEvent) {
  const payload = {
    ...event,
    timestamp: event.timestamp ?? new Date().toISOString(),
  };

  if (process.env.NEXT_PUBLIC_ANALYTICS_DEBUG === "true") {
    console.info("[analytics:client]", payload);
  }

  try {
    const body = JSON.stringify(payload);

    if (navigator.sendBeacon) {
      const blob = new Blob([body], { type: "application/json" });
      navigator.sendBeacon("/api/analytics", blob);
      return;
    }

    await fetch("/api/analytics", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body,
      keepalive: true,
    });
  } catch {
    // Swallow analytics errors to avoid affecting UX.
  }
}

    }
  }
}

import { getDiagnosticConfig } from "@/lib/content";

export type DiagnosticAnswerMap = Record<string, number>;

export type DiagnosticResult = {
  totalScore: number;
  maxScore: number;
  maturityTierId: string;
  maturityTierLabel: string;
  recommendedPath: string;
  recommendedProductSlug: string;
  recommendedFreeToolSlug: string;
  next7DayAction: string;
};

export async function scoreDiagnostic(answers: DiagnosticAnswerMap): Promise<DiagnosticResult> {
  const config = await getDiagnosticConfig();

  const totalScore = config.questions.reduce((sum, question) => {
    const value = answers[question.id];
    if (typeof value !== "number") {
      return sum;
    }
    const safeScore = Math.max(0, Math.min(3, Math.floor(value)));
    return sum + safeScore;
  }, 0);

  const maxScore = config.questions.length * 3;

  const tier =
    config.tiers.find(
      (item) => totalScore >= item.minScore && totalScore <= item.maxScore,
    ) ?? config.tiers[0];

  return {
    totalScore,
    maxScore,
    maturityTierId: tier.id,
    maturityTierLabel: tier.label,
    recommendedPath: tier.recommendedPath,
    recommendedProductSlug: tier.recommendedProductSlug,
    recommendedFreeToolSlug: tier.recommendedFreeToolSlug,
    next7DayAction: tier.next7DayAction,
  };
}
